    <script>
        const API_BASE_URL = 'http://localhost:5000';
        let isConnected = false;

        // Check API connection on load
        async function checkConnection() {
            try {
                const response = await fetch(`${API_BASE_URL}/health`);
                const data = await response.json();
                
                if (data.status === 'healthy') {
                    isConnected = true;
                    document.getElementById('connectionStatus').textContent = 'Connected to prediction service';
                    document.getElementById('statusDot').style.background = '#48bb78';
                } else {
                    throw new Error('Service unhealthy');
                }
            } catch (error) {
                isConnected = false;
                document.getElementById('connectionStatus').textContent = 'Cannot connect to prediction service';
                document.getElementById('statusDot').style.background = '#f56565';
                showAlert('Cannot connect to backend API. Please ensure the Flask server is running on port 5000.', 'error');
            }
        }

        // Load predefined scenarios
        function loadScenario(type) {
            const scenarios = {
                normal: {
                    temperature: 22.0,
                    smoke: 15.0,
                    co2: 550.0,
                    humidity: 50.0,
                    occupancy: 40,
                    hvac: 'On'
                },
                high_risk: {
                    temperature: 35.0,
                    smoke: 150.0,
                    co2: 1200.0,
                    humidity: 30.0,
                    occupancy: 20,
                    hvac: 'Off'
                },
                emergency: {
                    temperature: 45.5,
                    smoke: 750.0,
                    co2: 2800.0,
                    humidity: 25.0,
                    occupancy: 5,
                    hvac: 'Off'
                }
            };

            const scenario = scenarios[type];
            if (scenario) {
                document.getElementById('temperature').value = scenario.temperature;
                document.getElementById('smoke').value = scenario.smoke;
                document.getElementById('co2').value = scenario.co2;
                document.getElementById('humidity').value = scenario.humidity;
                document.getElementById('occupancy').value = scenario.occupancy;
                document.getElementById('hvac').value = scenario.hvac;
            }
        }

        // Show alert messages
        function showAlert(message, type) {
            const existingAlert = document.querySelector('.alert');
            if (existingAlert) {
                existingAlert.remove();
            }

            const alert = document.createElement('div');
            alert.className = `alert alert-${type}`;
            alert.textContent = message;
            
            const form = document.getElementById('predictionForm');
            form.parentNode.insertBefore(alert, form);

            setTimeout(() => {
                if (alert.parentNode) {
                    alert.remove();
                }
            }, 5000);
        }

        // Create risk visualization
        function createRiskVisualization(probability) {
            const percentage = Math.round(probability * 100);
            const color = percentage >= 60 ? '#f56565' : percentage >= 40 ? '#ed8936' : '#48bb78';
            
            return `
                <div style="text-align: center;">
                    <div style="position: relative; width: 150px; height: 150px; margin: 0 auto 20px;">
                        <svg width="150" height="150" style="transform: rotate(-90deg);">
                            <circle cx="75" cy="75" r="65" stroke="#e2e8f0" stroke-width="10" fill="none"></circle>
                            <circle cx="75" cy="75" r="65" stroke="${color}" stroke-width="10" fill="none" 
                                    stroke-dasharray="408.4" stroke-dashoffset="${408.4 - (408.4 * probability)}"
                                    style="transition: stroke-dashoffset 2s ease;"></circle>
                        </svg>
                        <div style="position: absolute; top: 50%; left: 50%; transform: translate(-50%, -50%); font-size: 2rem; font-weight: bold; color: ${color};">
                            ${percentage}%
                        </div>
                    </div>
                    <h3 style="color: ${color}; margin-bottom: 10px;">Fire Risk Probability</h3>
                    <p style="color: #666; font-size: 0.9rem;">Based on current sensor readings</p>
                </div>
            `;
        }

        // Handle form submission
        document.getElementById('predictionForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            
            if (!isConnected) {
                showAlert('Cannot make prediction - not connected to service', 'error');
                return;
            }

            const button = document.getElementById('predictBtn');
            const originalText = button.textContent;
            button.innerHTML = '<span class="loading"></span> Analyzing...';
            button.disabled = true;

            try {
                const formData = {
                    Temperature_C: parseFloat(document.getElementById('temperature').value),
                    Smoke_Level_ppm: parseFloat(document.getElementById('smoke').value),
                    CO2_Level_ppm: parseFloat(document.getElementById('co2').value),
                    Humidity_percent: parseFloat(document.getElementById('humidity').value),
                    Occupancy_Count: parseInt(document.getElementById('occupancy').value),
                    HVAC_Status: document.getElementById('hvac').value
                };

                const response = await fetch(`${API_BASE_URL}/predict`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(formData)
                });

                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }

                const result = await response.json();
                displayResults(result);
                
                // Update risk visualization
                document.getElementById('riskVisualization').innerHTML = createRiskVisualization(result.fire_probability);

            } catch (error) {
                console.error('Prediction error:', error);
                showAlert(`Prediction failed: ${error.message}`, 'error');
            } finally {
                button.textContent = originalText;
                button.disabled = false;
            }
        });

        // Display prediction results
        function displayResults(result) {
            const resultCard = document.getElementById('resultCard');
            const resultContent = document.getElementById('resultContent');
            
            let resultClass, iconClass;
            if (result.risk_level === 'CRITICAL' || result.risk_level === 'HIGH') {
                resultClass = 'result-danger';
                iconClass = 'üö®';
            } else if (result.risk_level === 'MEDIUM') {
                resultClass = 'result-warning';
                iconClass = '‚ö†Ô∏è';
            } else {
                resultClass = 'result-safe';
                iconClass = '‚úÖ';
            }

            const probabilityPercent = (result.fire_probability * 100).toFixed(1);
            
            resultContent.innerHTML = `
                <div class="result-content ${resultClass}">
                    <div style="text-align: center; margin-bottom: 20px;">
                        <div style="font-size: 3rem; margin-bottom: 10px;">${iconClass}</div>
                        <h3 style="font-size: 1.5rem; margin-bottom: 10px;">${result.risk_level} RISK</h3>
                        <p style="font-size: 1.1rem; margin-bottom: 10px;">${result.message}</p>
                        <p style="font-size: 2rem; font-weight: bold;">${probabilityPercent}% Fire Probability</p>
                    </div>
                    
                    <div class="result-grid">
                        <div class="result-item">
                            <h4>Temperature</h4>
                            <p>${result.input_data.Temperature_C}¬∞C</p>
                        </div>
                        <div class="result-item">
                            <h4>Smoke Level</h4>
                            <p>${result.input_data.Smoke_Level_ppm} ppm</p>
                        </div>
                        <div class="result-item">
                            <h4>CO2 Level</h4>
                            <p>${result.input_data.CO2_Level_ppm} ppm</p>
                        </div>
                        <div class="result-item">
                            <h4>Humidity</h4>
                            <p>${result.input_data.Humidity_percent}%</p>
                        </div>
                        <div class="result-item">
                            <h4>Occupancy</h4>
                            <p>${result.input_data.Occupancy_Count} people</p>
                        </div>
                        <div class="result-item">
                            <h4>HVAC Status</h4>
                            <p>${result.input_data.HVAC_Status}</p>
                        </div>
                    </div>
                </div>
            `;
            
            resultCard.style.display = 'block';
            resultCard.scrollIntoView({ behavior: 'smooth' });
        }

        // Initialize the application
        window.addEventListener('load', function() {
            checkConnection();
            
            // Update risk visualization with default values
            document.getElementById('riskVisualization').innerHTML = createRiskVisualization(0.1);
            
            // Retry connection every 30 seconds if disconnected
            setInterval(() => {
                if (!isConnected) {
                    checkConnection();
                }
            }, 30000);
        });
    </script>
